const circle = document.querySelector('.circle') function colorChange (e) { if (e.key === 'a') { circle.style.backgroundColor = 'blue' } else if (e.key === 's') { circle.style.backgroundColor = 'green' } } function colorReset () { circle.style.backgroundColor = "yellow" } //window.addEventListener('keypress', colorChange) window.addEvenListener('keydown', colorChange) window.addEvenListener('keyup', colorReset)
